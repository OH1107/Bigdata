  ```sql
  -- 가장 많이 받는 급여
  select max(salary) from salaries where to_date>curdate();
  -- 급여를 많이 받는 사원의 번호
  select emp_no from salaries where salary=(select max(salary) from salaries) and to_date>curdate();
  ```
